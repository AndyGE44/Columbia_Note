\section*{1. General Dynamic Programming Concepts}
\begin{itemize}
    \item \textbf{Optimal Substructure:} An optimal solution contains optimal solutions to subproblems.
    \item \textbf{Overlapping Subproblems:} Subproblems are solved repeatedly; DP stores results to avoid re-computation.
    \item \textbf{Approach:} Define subproblem $\rightarrow$ Recursive formula $\rightarrow$ Bottom-up computation $\rightarrow$ Construct solution.
\end{itemize}

\hrulefill

\section*{2. Rod Cutting}
\textbf{Problem:} Maximize revenue $r_n$ for a rod of length $n$ given prices $p_i$.
\begin{itemize}
    \item \textbf{Recurrence Formula:}
    \[
    r_n = \max_{1 \le i \le n} (p_i + r_{n-i}), \quad \text{with } r_0 = 0
    \]
    \item \textbf{Meaning:} We cut a piece of length $i$ first, then optimally solve for the remaining length $n-i$.
    \item \textbf{Time Complexity:} $\Theta(n^2)$.
    \item \textbf{Reconstruction:} Store $k[j]$ (the optimal first cut size for length $j$). Print $k[n]$, then recurse on $n - k[n]$.
\end{itemize}

\hrulefill

\section*{3. Longest Common Subsequence (LCS)}
\textbf{Problem:} Find length of longest subsequence present in both $X=(x_1...x_m)$ and $Y=(y_1...y_n)$.
\begin{itemize}
    \item \textbf{State:} $c[i, j]$ = LCS length for prefixes $X_i$ and $Y_j$.
    \item \textbf{Recurrence Formula:}
    \[
    c[i,j] = 
    \begin{cases} 
    0 & \text{if } i=0 \text{ or } j=0 \\
    c[i-1, j-1] + 1 & \text{if } x_i = y_j \\
    \max(c[i, j-1], c[i-1, j]) & \text{if } x_i \neq y_j 
    \end{cases}
    \]
    \item \textbf{Time Complexity:} $\Theta(m \cdot n)$.
    \item \textbf{Reconstruction:} Start at $c[m,n]$. If $x_i = y_j$, go diagonal ($\nwarrow$) and print char. If $x_i \neq y_j$, move to the larger neighbor ($\uparrow$ or $\leftarrow$).
\end{itemize}

\hrulefill

\section*{4. Optimal Binary Search Tree (Optimal BST)}
\textbf{Problem:} Build a BST for keys $K=\{k_1...k_n\}$ with probabilities $p_i$ to minimize expected search cost.
\begin{itemize}
    \item \textbf{Cost Definition:} $\text{Cost}(T) = \sum_{i=1}^{n} p_i \cdot (\text{depth}(k_i) + 1)$.
    \item \textbf{State:} $c[i, j]$ = min cost for optimal BST containing keys $k_i \dots k_j$.
    \item \textbf{Recurrence Formula:}
    \[
    c[i,j] = 
    \begin{cases} 
    0 & \text{if } j = i-1 \text{ (empty tree)} \\
    p_{i,j} + \min_{i \le r \le j} \{ c[i, r-1] + c[r+1, j] \} & \text{if } i \le j 
    \end{cases}
    \]
    \item \textbf{Note:} $p_{i,j} = \sum_{k=i}^j p_k$ is added because choosing a root increases the depth of all sub-nodes by 1.
    \item \textbf{Time Complexity:} $\Theta(n^3)$. (Three nested loops: length, start index, root index).
    \item \textbf{Calculation Order:} By increasing length of interval ($len=1 \to n$).
\end{itemize}
